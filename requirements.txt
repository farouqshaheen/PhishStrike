# PhishStrike - Python Dependencies
# Install with: pip install -r requirements.txt
#
# Windows:
#   pip install -r requirements.txt
#
# Linux/macOS (if pip3):
#   pip3 install -r requirements.txt
#
# Kali Linux / Debian systems (if externally-managed error):
#   pip3 install -r requirements.txt --break-system-packages
#
# Termux (Android):
#   pip install -r requirements.txt

# ─── Core QR Code Generation ────────────────────────────────────────────
qrcode[pil]        # QR code generator (requires Pillow for image output)
Pillow             # Image processing library (required by qrcode[pil])

# ─── Web Dashboard (Flask-based live panel) ─────────────────────────────
Flask              # Lightweight web framework for the dashboard
Flask-SocketIO     # Real-time WebSocket support for live credential feed

# ─── AI Phishing Assistant ──────────────────────────────────────────────
google-generativeai # Gemini AI API integration for generating phishing templates

# ─── Data Export (Excel / PDF / CSV) ────────────────────────────────────
pandas             # Data manipulation and Excel export
openpyxl           # Excel (.xlsx) file writer (required by pandas for Excel)
fpdf2              # PDF report generation

# ─── Networking & HTTP ──────────────────────────────────────────────────
requests           # HTTP requests for URL checking and API calls
